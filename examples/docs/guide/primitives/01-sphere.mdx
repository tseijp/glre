export const frag = `
precision highp float;
uniform vec2 iResolution; // canvas size
uniform vec3 up;          // camera up direction
uniform vec3 eye;         // camera position
uniform vec3 focus;       // camera focus point
uniform float focal;      // camera focal length
uniform float size;       // object size
/**
 * sphereSDF
 */
float sphereSDF(vec3 p, float s) {
        return length(p) - s;
}
/**
 * main
 */
void main() {
        vec3 look = normalize(focus - eye);
        vec3 right = normalize(cross(look, up));
        vec2 scr = gl_FragCoord.xy - .5 * iResolution;
        vec3 dir = normalize(focal * look + scr.x * right + scr.y * up);
        /**
         * Ray marching
         */
        vec3 p = eye + dir;
        vec3 e = vec3(.0005, 0., 0.);
        float d = sphereSDF(p, size);
        for (int i = 0; i < 50; i++) {
                if (d <= e.x) {
                        float dx = sphereSDF(p + e.xyy, size) - d;
                        float dy = sphereSDF(p + e.yxx, size) - d;
                        float dz = sphereSDF(p + e.yyx, size) - d;
                        vec3 norm = normalize(vec3(dx, dy, dz));
                        gl_FragColor = vec4(norm * .5 + .5, 1.);
                        return;
                }
                p = p + d * normalize(dir);
                d = sphereSDF(p, size);
        }
}`

import { Leva } from 'leva'
import {
        useGLRender,
        useResizeRef,
        useOrbitControls,
        useUniformControls,
} from './hooks'

export const Canvas = () => {
        const self = useGLRender(frag)
        const ref = useResizeRef(self)
        useOrbitControls('eye', self)
        useUniformControls(
                {
                        up: [0, 1, 0],
                        focus: [0, 0, 0],
                        focal: 500,
                        size: 50,
                },
                self
        )
        return (
                <div ref={ref}>
                        <canvas ref={self.ref} />
                        <Leva />
                </div>
        )
}
