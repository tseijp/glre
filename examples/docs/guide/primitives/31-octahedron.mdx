export const frag = `
precision highp float;
uniform vec2 iResolution; // canvas size
uniform vec3 up;          // camera up direction
uniform vec3 eye;         // camera position
uniform vec3 focus;       // camera focus point
uniform float focal;      // camera focal length
uniform float size;
/**
 * SDF
 */
float octahedronSDF(vec3 p, float s) {
        p = abs(p);
        float m = p.x + p.y + p.z - s;
        vec3 q;
        if (3. * p.x < m) q = p.xyz;
        else if ( 3. * p.y < m ) q = p.yzx;
        else if ( 3. * p.z < m ) q = p.zxy;
        else return m * .57735027;
        float k = clamp(.5 * (q.z - q.y + s), .0, s);
        return length(vec3(q.x, q.y - s + k,q.z - k));
}
float SDF (vec3 p) {
        return octahedronSDF(p, size);
}
/**
 * main
 */
void main() {
        vec3 look = normalize(focus - eye);
        vec3 right = normalize(cross(look, up));
        vec2 scr = gl_FragCoord.xy - iResolution * .5;
        vec3 dir = normalize(focal * look + scr.x * right + scr.y * up);
        /**
         * Ray marching
         */
        vec3 p = eye + dir;
        vec3 e = vec3(.0005, 0., 0.);
        float d = SDF(p);
        for (int i = 0; i < 50; i++) {
                if (d <= e.x) {
                        float dx = SDF(p + e.xyy) - d;
                        float dy = SDF(p + e.yxy) - d;
                        float dz = SDF(p + e.yyx) - d;
                        vec3 norm = normalize(vec3(dx, dy, dz));
                        gl_FragColor = vec4(norm * .5 + .5, 1.);
                        return;
                }
                p = p + d * dir;
                d = SDF(p);
        }
}`

import { Leva } from 'leva'
import {
        useGLRender,
        useResizeRef,
        useOrbitControls,
        useUniformControls,
} from './hooks'

export const Canvas = () => {
        const self = useGLRender(frag)
        const ref = useResizeRef(self)
        useOrbitControls('eye', self)
        useUniformControls(
                {
                        up: [0, 1, 0],
                        focus: [0, 0, 0],
                        focal: 500,
                        size: 50,
                },
                self
        )
        return (
                <div ref={ref}>
                        <canvas ref={self.ref} />
                        <Leva />
                </div>
        )
}
